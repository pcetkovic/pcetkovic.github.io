GDP Growth Dashboard

d3 = require("d3@7")

// --- Variablen ---
margin = ({ top: 20, right: 50, bottom: 30, left: 50 })
small = ({ width: 200, height: 100 })
large = ({ width: 400, height: 200 })

// --- Beispiel-Daten ---
gdpData = [
{ year: new Date(2015,0,1), value: 2.5 },
{ year: new Date(2016,0,1), value: 1.8 },
{ year: new Date(2017,0,1), value: 2.9 },
{ year: new Date(2018,0,1), value: 2.4 },
{ year: new Date(2019,0,1), value: 2.3 },
{ year: new Date(2020,0,1), value: -3.4 },
{ year: new Date(2021,0,1), value: 5.7 },
{ year: new Date(2022,0,1), value: 2.1 },
{ year: new Date(2023,0,1), value: 1.9 },
{ year: new Date(2024,0,1), value: 2.2 }
]

inflationData = [
{ year: new Date(2015,0,1), value: 0.1 },
{ year: new Date(2016,0,1), value: 1.3 },
{ year: new Date(2017,0,1), value: 2.1 },
{ year: new Date(2018,0,1), value: 2.4 },
{ year: new Date(2019,0,1), value: 1.8 },
{ year: new Date(2020,0,1), value: 0.7 },
{ year: new Date(2021,0,1), value: 4.7 },
{ year: new Date(2022,0,1), value: 8.0 },
{ year: new Date(2023,0,1), value: 3.5 },
{ year: new Date(2024,0,1), value: 2.5 }
]

// --- Funktion zur Erstellung eines Diagramms ---
createChart = (data, color, label) => {
const svg = d3.select(DOM.svg(small.width, small.height))
.attr("viewBox", `0 0 ${small.width} ${small.height}`)
//.style("border", "1px solid #ccc")
.style("margin-right", "20px")
.style("cursor", "pointer")

const g = svg.append("g")
             .attr("transform", `translate(${margin.left},${margin.top})`)

const innerWidth = small.width - margin.left - margin.right
const innerHeight = small.height - margin.top - margin.bottom

const x = d3.scaleTime() .domain(d3.extent(data, d => d.year)) .range([0, innerWidth])
const y = d3.scaleLinear() .domain([d3.min(data, d=>d.value)-1, d3.max(data,d=>d.value)+1]) .range([innerHeight, 0])

//g.append("g")
// .attr("transform", `translate(0,${innerHeight})`)
// .call(d3.axisBottom(x).ticks(5))

//g.append("g")
// .call(d3.axisLeft(y).ticks(5))

const line = d3.line()
               .x(d => x(d.year))
               .y(d => y(d.value))

g.append("path")
 .datum(data)
 .attr("fill", "none")
 .attr("stroke", color)
 .attr("stroke-width", 2)
 .attr("d", line)

const circles = g.selectAll("circle")
                 .data(data)
                 .enter()
                 .append("circle")
                 .attr("cx", d => x(d.year))
                 .attr("cy", d => y(d.value))
                 .attr("r", 4)
                 .attr("fill", color)

// --- Tooltip ---
const tooltip = d3.select(document.body)
                  .append("div")
                  .style("position", "absolute")
                  .style("padding", "6px 10px")
                  .style("background", "rgba(0,0,0,0.7)")
                  .style("color", "#fff")
                  .style("border-radius", "4px")
                  .style("pointer-events", "none")
                  .style("opacity", 0)
                  .style("font-size", "12px")

circles.on("mouseover", (event,d) => {
         tooltip.style("opacity", 1)
                .html(`<strong>${label}:</strong> ${d.value}%<br><strong>Year:</strong> ${d.year.getFullYear()}`)
         d3.select(event.currentTarget).transition().duration(200).attr("r", 7)
      })
       .on("mousemove", event => {
         tooltip.style("left", (event.pageX + 10) + "px")
                .style("top", (event.pageY - 25) + "px")
       })
       .on("mouseout", event => {
         tooltip.style("opacity", 0)
         d3.select(event.currentTarget).transition().duration(200).attr("r", 4)
       })

// --- Hover-Vergrößerung ---
svg.style("overflow", "visible")

svg.on("mouseover", () => {
     g.transition().duration(300)
      .style("transform-origin", "0 0") // oder "top left"
      .style("transform", `scale(${large.width/small.width}, ${large.height/small.height})`)
   })
   .on("mouseout", () => {
     g.transition().duration(300)
      .style("transform", "scale(1)")
   })

return svg.node()




}




// HTML-Wrapper direkt in OJS zurückgeben
html`<div style="display:inline-block; position:relative; overflow:visible;">
  ${createChart(gdpData, "steelblue", "GDP Growth")}
  ${createChart(inflationData, "orange", "Inflation")}
</div>`
